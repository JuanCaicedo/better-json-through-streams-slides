Data in JSON

Bad client connection

Our game plan

  • One request
  • Stream server -> client
  • One stream for all data

Toolkit

  • Highland.js

Push points to stream

              
/* points-module */
function convertToPointStream(sourceStream) {

  return highland(function(push) {

    oboe(sourceStream)
      .node('{x y color}', function(point) {
        push(null, point);
      })

  });

}

Terminate stream

              
/* points-module */
function convertToPointStream(sourceStream) {

  return highland(function(push) {

    oboe(sourceStream)
      .node('{x y color}', function(point) {
        push(null, point);
      })
      .done(function() {
        push(null, highland.nil);
      });

  });

}

Unify streams!

              
/* points-module */
function getDataStream() {

  var catPath = path.resolve(__dirname, './cat-points.json');
  var catSource = fs.createReadStream(catPath);
  var catStream = convertToPointStream(catSource);

  var sunUrl = 'urlFromGithub';
  var sunSource = request(sunUrl);
  var sunStream = convertToPointStream(sunSource);

  return highland([
    catStream,
    sunStream
  ])
  .merge();

}
